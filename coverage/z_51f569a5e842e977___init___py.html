35""" 

36.. _python_interface: 

37 

38GHDL offers two Python interfaces and a language server protocol service. All 

39this is provided from a ``pyGHDL`` packages with four sub-packages: 

40 

41* :mod:`pyGHDL.cli` - Command line interface (CLI) applications. 

42* :mod:`pyGHDL.dom` - A high-level API offering a document object model (DOM). 

43 The underlying abstract VHDL language model is provided by :doc:`pyVHDLModel <vhdlmodel:index>`. 

44 The DOM is using ``libghdl`` for file analysis and parsing. 

45* :mod:`pyGHDL.libghdl` - A low-level API directly interacting with the shared library ``libghdl....so``/``libghdl....dll``. 

46 This is a procedural and C-like interface. It comes with some Python generators for easier iterating linked lists. 

47* :mod:`pyGHDL.lsp` - A :wikipedia:`language server protocol <Language_Server_Protocol>` (LSP) 

48 written in Python. The implementation offers an HTTPS service that can be used e.g. by editors and IDEs supporting LSP. 

49"""

66@export 

67class GHDLBaseException(Exception): 

68 """Base exception derived from :exc:`Exception <python:Exception>` for all custom exceptions.""" 

69 

70 _message: str 

71 

72 def __init__(self, message: str = "") -> None: 

73 """ 

74 GHDLBaseException initializer. 

75 

76 :param message: The exception message. 

77 """ 

78 super().__init__() 

79 self._message = message 

80 

81 # WORKAROUND: for Python <3.11 

82 # Implementing a dummy method for Python versions before 

83 if version_info < (3, 11): # pragma: no cover 

84 __notes__: List[str] #: A list of notes for this exception. 

85 

86 def add_note(self, message: str) -> None: 

87 """ 

88 Add a note to the exception. 

89 

90 :param message: The note's message. 

91 """ 

92 try: 

93 self.__notes__.append(message) 

94 except AttributeError: 

95 self.__notes__ = [message] 

96 

97 @readonly 

98 def message(self) -> str: 

99 """ 

100 Returns the exception's message. 

101 

102 :returns: Exception message. 

103 """ 

104 return self._message 

105 

106 def __str__(self) -> str: 

107 """ 

108 Returns the exception's message. 

109 

110 :returns: Exception message. 

111 """ 

112 return self._message
